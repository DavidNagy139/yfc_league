⚙️ إعداد Supabase

يحتاج النظام إلى ربطه بقاعدة بيانات Supabase. أدخل بياناتك مرة واحدة وسيتذكرها النظام.

خطوات الإعداد:

  1. افتح supabase.com وأنشئ مشروع جديد
  2. من Settings → API انسخ Project URL
  3. انسخ anon public key
  4. شغّل الـ SQL أدناه في SQL Editor

📋 SQL لإنشاء الجداول – اضغط هنا لنسخه

create table teams (id uuid default gen_random_uuid() primary key, name text not null unique, abbr text, created_at timestamptz default now());

create table matches (id uuid default gen_random_uuid() primary key, team1 text not null, team2 text not null, round int, match_date date, match_time time, played boolean default false, goals1 int default 0, goals2 int default 0, scorers text, created_at timestamptz default now());

create table scorers (id uuid default gen_random_uuid() primary key, player_name text not null, team_name text, goals int default 0);

alter table teams enable row level security;
alter table matches enable row level security;
alter table scorers enable row level security;
create policy "public read teams" on teams for select using (true);
create policy "public read matches" on matches for select using (true);
create policy "public read scorers" on scorers for select using (true);
create policy "auth write teams" on teams for all using (auth.role() = 'authenticated');
create policy "auth write matches" on matches for all using (auth.role() = 'authenticated');
create policy "auth write scorers" on scorers for all using (auth.role() = 'authenticated');
